##Setup Event - Sets flags for option decisions at game start

	id = 10000000
	title = "EVTNAME10000000"
	desc = "EVTDESC10000000"
	picture = "TGC_logo"
	
	fire_only_once = yes

	trigger = {
		year = 1836
		ai = no
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 1
			year = 1836
		}
	}
	
	#Singleplayer
	option = {
		name = "EVTOPTA10000000"
		set_country_flag = show_options_singleplayer
		ai_chance = { factor = 0 }
	}
	
	#Multiplayer - Host
	option = {
		name = "EVTOPTB10000000"
		set_country_flag = show_options_multiplayer_host
		ai_chance = { factor = 0 }
	}
	
	#Multiplayer - Standard
	option = {
		name = "EVTOPTC10000000"
		set_country_flag = show_options_multiplayer_standard
		ai_chance = { factor = 1 }
	}
}
